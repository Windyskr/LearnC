### 习题2-4 韩信点兵-hanxin 
#### 方法1 穷举
可怜可怜，只会用穷举。
```
#include <stdio.h>
#define MIN 10
#define MAX 100
int hanxin(int a, int b, int c) {
    int i;
    for (i = MIN; i < MAX; i++) {
        if ((i % 3 == a) && (i % 5 == b) && (i % 7 == c)) {
            return i;
        }
    }
    return 0;
}
int main() {
    int sum, a, b, c, kase = 0;
    while ((scanf("%d", &a)) && (scanf("%d", &b)) && (scanf("%d", &c))) {
        printf("Case %d:", ++kase);
        if ((sum = hanxin(a, b, c)) != 0) {
            printf("%d", sum);
        } else {
            printf("No answer");
        }
        printf("\n");
    }
    return 0;
}
```

### 习题2-3 倒三角型-triangle 
和我们课程测试题目差不多，甚至更简单。
```
#include <stdio.h>
void space(int x) {
    while (x--) {
        printf(" ");
    }
}
void jin(int x) {
    while (x--) {
        printf("#");
    }
}
int main() {
    int n;
    int i = 0;
    scanf("%d", &n);
    if (n <= 20 && n > 0) {
        while (n) {
            space(i++);
            jin(2 * (n--) - 1);
            printf("\n");
        }
    }
    return 0;
}
```

### 习题2-4 子序列的和-subsequence
```
#include <stdio.h>
double cal(long n, long m) {
    double sum = 0;
    for (n = n - 1; n < m; n++) {
        sum += 1 / (double)((n + 1) * (n + 1));
    }
    return sum;
}
int main() {
    int kase = 0;
    long n, m;
    while ((scanf("%ld%ld", &n, &m)) && ((n != 0) && (m != 0))) {
        if (kase) printf("\n");
        printf("Case %d:%.5f", ++kase, cal(n, m));
    }
    return 0;
}
```
